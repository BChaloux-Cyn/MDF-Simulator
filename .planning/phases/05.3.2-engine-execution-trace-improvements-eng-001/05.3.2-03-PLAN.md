```python
def test_classify_queue_subtype(fast_clock):
    """D-12: _classify_queue distinguishes cross-class events even when IDs match.

    Specifically verifies CarDoor(door_id=1) -> ShaftDoor(door_id=1) is classified
    as 'standard', NOT priority. The sender_id dict equals the target_id dict,
    but sender_class != target_class, so the event is cross-instance.
    """
    from engine.registry import InstanceRegistry
    from engine.scheduler import ThreeQueueScheduler
    from engine.event import Event

    # Minimal manifest with CarDoor and ShaftDoor (no real behaviour needed —
    # we only call _classify_queue which inspects the Event, not the registry).
    class_defs = {
        "CarDoor": {
            "name": "CarDoor",
            "is_abstract": False,
            "identifier_attrs": ["door_id"],
            "attributes": {"door_id": "int"},
            "initial_state": "Closed",
            "final_states": [],
            "transition_table": {},
            "supertype": "Door",
            "subtypes": [],
        },
        "ShaftDoor": {
            "name": "ShaftDoor",
            "is_abstract": False,
            "identifier_attrs": ["door_id"],
            "attributes": {"door_id": "int"},
            "initial_state": "Closed",
            "final_states": [],
            "transition_table": {},
            "supertype": "Door",
            "subtypes": [],
        },
    }
    registry = InstanceRegistry(class_defs)
    scheduler = ThreeQueueScheduler(registry, fast_clock, class_defs)

    # Sub-case A: self-directed (same class, same id) -> priority
    self_evt = Event(
        event_type="Tick",
        sender_class="CarDoor", sender_id={"door_id": 1},
        target_class="CarDoor", target_id={"door_id": 1},
    )
    assert scheduler._classify_queue(self_evt) == "priority"

    # Sub-case B (D-12 target): cross-class, same id -> standard
    cross_evt = Event(
        event_type="Door_open",
        sender_class="CarDoor", sender_id={"door_id": 1},
        target_class="ShaftDoor", target_id={"door_id": 1},
    )
    assert scheduler._classify_queue(cross_evt) == "standard", (
        "CarDoor -> ShaftDoor with matching door_id must be 'standard' "
        "(different classes), but was classified as priority. "
        "_classify_queue must check sender_class == target_class AND sender_id == target_id."
    )

    # Sub-case C: same class, different id -> standard
    other_evt = Event(
        event_type="Tick",
        sender_class="CarDoor", sender_id={"door_id": 1},
        target_class="CarDoor", target_id={"door_id": 2},
    )
    assert scheduler._classify_queue(other_evt) == "standard"

    # Sub-case D: __creation__ is always standard regardless of routing
    creation_evt = Event(
        event_type="__creation__",
        sender_class="CarDoor", sender_id={"door_id": 1},
        target_class="CarDoor", target_id={"door_id": 1},
    )
    assert scheduler._classify_queue(creation_evt) == "standard"
```
