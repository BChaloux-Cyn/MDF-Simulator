        _step_counter: int = 0
        _start_ms: float = time.monotonic() * 1000.0

        def _stamp(step: MicroStep) -> MicroStep:
            nonlocal _step_counter
            ts = time.monotonic() * 1000.0 - _start_ms
            stamped = dataclasses.replace(step, step_id=_step_counter, timestamp_ms=ts)

    from engine.scheduler import ThreeQueueScheduler

    steps = list(scheduler.execute())
