```python
def _render_action_stmts(
    fn_name_prefix: str,
    body_src: str,
    source_file: str,
    source_line: int,
) -> str:
    """Render a D-02 list-of-tuples action slot.

    Emits:
        def _stmt_0_<prefix>(ctx, self_dict, params): ...
        def _stmt_1_<prefix>(ctx, self_dict, params): ...
        <prefix> = [("<pycca_src_0>", _stmt_0_<prefix>), ("<pycca_src_1>", _stmt_1_<prefix>), ...]

    For empty/whitespace body_src (D-06):
        def _stmt_0_<prefix>(ctx, self_dict, params): pass
        <prefix> = [("pass", _stmt_0_<prefix>)]

    Each per-statement callable has the same signature as the old monolithic
    action_fn: (ctx: SimulationContext, self_dict: dict, params: dict) -> None.
    The scheduler iterates the list, calls each in order, and yields one
    ActionExecuted per tuple (D-02/D-05/D-06).
    """
    comment = f"# from {source_file}:{source_line}"
    if not body_src or not body_src.strip():
        # D-06: empty body → single pass sentinel
        return (
            f"{comment}\n"
            f"def _stmt_0_{fn_name_prefix}(ctx: \"SimulationContext\", self_dict: dict, params: dict) -> None:\n"
            f"    pass\n"
            f"{fn_name_prefix} = [(\"pass\", _stmt_0_{fn_name_prefix})]\n"
        )

    parser = make_action_parser()
    try:
        tree = parser.parse(body_src)
    except Exception as exc:
        # Soft-fail policy matches _render_action_fn: emit single pass sentinel
        # with a warning, so the bundle remains loadable.
        warnings.warn(
            f"Failed to parse action for {fn_name_prefix}: {exc}",
            stacklevel=2,
        )
        return (
            f"{comment}\n"
            f"def _stmt_0_{fn_name_prefix}(ctx: \"SimulationContext\", self_dict: dict, params: dict) -> None:\n"
            f"    pass\n"
            f"{fn_name_prefix} = [(\"pass\", _stmt_0_{fn_name_prefix})]\n"
        )

    fn_defs: list[str] = []
    pair_exprs: list[str] = []
    for idx, stmt_tree in enumerate(tree.children):
        start = stmt_tree.meta.start_pos
        end = stmt_tree.meta.end_pos
        pycca_src = body_src[start:end].strip() or "pass"
        transformer = ActionTransformer()
        try:
            py_body = transformer.transform(stmt_tree)
        except Exception as exc:
            warnings.warn(
                f"Failed to transform statement {idx} of {fn_name_prefix}: {exc}",
                stacklevel=2,
            )
            py_body = "pass"
        indented = textwrap.indent(py_body.strip() or "pass", "    ")
        fn_label = f"_stmt_{idx}_{fn_name_prefix}"
        fn_defs.append(
            f"def {fn_label}(ctx: \"SimulationContext\", self_dict: dict, params: dict) -> None:\n"
            f"{indented}\n"
        )
        pair_exprs.append(f"    ({pycca_src!r}, {fn_label})")

    pairs_block = ",\n".join(pair_exprs)
    return (
        f"{comment}\n"
        + "\n".join(fn_defs)
        + f"\n{fn_name_prefix} = [\n{pairs_block},\n]\n"
    )
```
