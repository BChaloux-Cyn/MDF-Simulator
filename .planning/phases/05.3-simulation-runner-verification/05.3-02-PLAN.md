```python
def create(self, class_name: str, attrs: dict) -> dict:
    """Generated-code form. attrs must include identifier attributes.
    Splits attrs into identifier (per ClassManifest) and non-identifier fields,
    calls self.registry.create_sync(class_name, identifier, attrs=attrs),
    and returns the new instance dict (which already has __instance_key__ / __class_name__
    set by Task 1). Emits instance_created micro-step."""

def delete(self, inst: dict) -> None:
    """Generated-code form. Uses inst['__class_name__'] and inst['__instance_key__']
    to call self.registry.delete_sync. Emits instance_deleted micro-step."""

def delete_where(self, class_name: str, predicate) -> None:
    """Delete all instances of class_name matching predicate(inst) -> bool."""

def relate(self, a: dict, b: dict, rel_id: str) -> None:  # overload/replace
    """Generated-code form. Extracts class/key from dicts and calls link_store.relate.
    If the existing method has a different signature, add a new method name like
    `relate_by_dicts` and have the transformer call it — OR make relate a dispatch
    on argument types (if first arg is a dict, use new path; else old path)."""

def unrelate(self, a: dict, b: dict, rel_id: str) -> None: ...

def select_any(self, class_name: str, where=None) -> dict | None:
    """Return any instance of class_name matching where(inst) or any instance if where is None."""

def select_many(self, class_name: str, where=None) -> list[dict]: ...

def traverse(self, source: dict, rel_chain: list[str]) -> list[dict]:
    """Chain-navigate relationships. source is an instance dict; rel_chain is a list of
    relationship IDs. Uses __class_name__ + __instance_key__ to query link_store at each hop."""

def select_any_related(self, source: dict, rel_chain: list[str]) -> dict | None:
    result = self.traverse(source, rel_chain)
    return result[0] if result else None

def select_many_related(self, source: dict, rel_chain: list[str]) -> list[dict]:
    return self.traverse(source, rel_chain)

def bridge(self, domain: str, op: str, args: dict):
    """Look up (domain, op) in self.bridge_mocks and return the mocked value.
    Emit bridge_called micro-step. Return None for undefined operations (per MCP-08)."""

def cancel(self, event_name: str, sender: dict, target: dict) -> None:
    """Cancel a pending delayed event matching (event_name, sender_key, target_key) in the delay queue."""
```
