```python
"""Conditional trigger evaluation (D-18..D-21)."""
from dataclasses import dataclass, field
from schema.scenario_schema import TriggerDef

TRIGGER_FIRE_LIMIT = 10_000


@dataclass
class ArmedTrigger:
    definition: TriggerDef
    armed: bool = True
    fire_count: int = 0


class TriggerEvaluator:
    def __init__(self, triggers: list[TriggerDef], alias_to_instance: dict[str, dict]):
        """alias_to_instance maps scenario instance names to live instance dicts (populated after setup)."""
        self.armed = [ArmedTrigger(t) for t in triggers]
        self.aliases = alias_to_instance
        self.total_fires = 0

    def evaluate(self, ctx) -> list[TriggerDef]:
        """Return list of triggers that should fire now. Disarms non-repeat after selection."""
        to_fire: list[TriggerDef] = []
        for at in self.armed:
            if not at.armed:
                continue
            if self._matches(at.definition, ctx):
                to_fire.append(at.definition)
                at.fire_count += 1
                self.total_fires += 1
                if not at.definition.repeat:
                    at.armed = False
        if self.total_fires > TRIGGER_FIRE_LIMIT:
            raise RuntimeError(f"Trigger fire limit {TRIGGER_FIRE_LIMIT} exceeded — possible infinite loop")
        return to_fire

    def _matches(self, trig: TriggerDef, ctx) -> bool:
        cond = trig.when
        inst = self.aliases.get(cond.instance)
        if inst is None:
            return False
        if cond.state is not None:
            current = ctx.registry.current_state(inst["__class_name__"], inst["__instance_key__"])
            if current != cond.state:
                return False
        if cond.attr is not None:
            if inst.get(cond.attr) != cond.eq:
                return False
        return True
```

```python
"""Scenario runner: alias resolution, event dispatch, trigger interleaving."""
from schema.scenario_schema import ScenarioDef, EventDef, TriggerAction
from engine.trigger import TriggerEvaluator


def run_scenario(ctx, scenario: ScenarioDef, manifest: dict):
    """Generator yielding micro-steps.

    1. Create instances in ctx.registry, build alias -> instance_dict map
    2. Apply initial relationships via ctx.relate(a, b, rel)
    3. Enqueue events (respecting at_ms / after_ms timing) with sender/target resolved from aliases
    4. Drive ctx.execute(), evaluating triggers after each micro-step
    5. Trigger actions are posted as events/method calls in the same ctx
    """
    aliases: dict[str, dict] = {}

    # Step 1: create instances
    for inst_def in scenario.instances:
        inst = ctx.create(inst_def.class_name, {**inst_def.id, **inst_def.attrs})
        if inst_def.state is not None:
            ctx.registry.set_state(inst_def.class_name, inst["__instance_key__"], inst_def.state)
        aliases[inst_def.name] = inst

    # Step 2: initial relationships
    for rel in scenario.relationships:
        a = aliases[rel.a]
        b = aliases[rel.b]
        ctx.relate(a, b, rel.rel)

    # Step 3: enqueue scenario events with timing
    current_time = 0.0
    for ev in scenario.events:
        if ev.at_ms is not None:
            fire_time = ev.at_ms
        elif ev.after_ms is not None:
            fire_time = current_time + ev.after_ms
        else:
            fire_time = current_time
        current_time = fire_time
        target_inst = aliases[ev.target]
        sender_inst = aliases[ev.sender]
        if ev.event:
            ctx.generate(
                ev.event,
                target=target_inst["__instance_key__"],
                args=ev.args,
                sender=sender_inst,
                delay_ms=fire_time,
            )
        elif ev.call:
            _invoke_method(ctx, sender_inst, target_inst, ev.call, ev.args, manifest)

    # Step 4: drive execution with trigger evaluation
    evaluator = TriggerEvaluator(scenario.triggers, aliases)
    for step in ctx.execute():
        yield step
        for trig in evaluator.evaluate(ctx):
            _fire_trigger_action(ctx, trig.then, aliases, manifest)


def _fire_trigger_action(ctx, action: TriggerAction, aliases: dict, manifest: dict) -> None:
    target = aliases[action.target]
    sender = aliases[action.sender]
    if action.event:
        ctx.generate(action.event, target=target["__instance_key__"], args=action.args, sender=sender)
    elif action.call:
        _invoke_method(ctx, sender, target, action.call, action.args, manifest)


def _invoke_method(ctx, sender: dict, target: dict, method_name: str, args: dict, manifest: dict) -> None:
    """Execute the action body of a method directly (D-17)."""
    class_name = target["__class_name__"]
    class_def = manifest["class_defs"].get(class_name, {})
    methods = class_def.get("methods", {})
    method_fn = methods.get(method_name)
    if method_fn is None:
        raise RuntimeError(f"Method {class_name}.{method_name} not found in manifest")
    ctx._set_executing_instance(sender)
    try:
        method_fn(ctx, target, args)
    finally:
        ctx._set_executing_instance(None)
```

```python
"""MCP tool wrappers: simulate_domain, simulate_class (Phase 5.3)."""
import json
import re
import shutil
import sys
from dataclasses import asdict, is_dataclass
from datetime import datetime
from pathlib import Path

import yaml

from engine.bundle_loader import BundleCorruptError, BundleVersionError, load_bundle
from engine.ctx import SimulationContext
from engine.preflight import check_multiplicity
from engine.scenario_runner import run_scenario
from schema.scenario_schema import ScenarioDef

# MODEL_ROOT anchored to CWD (established Phase 02 pattern)
def _model_root() -> Path:
    return Path.cwd() / ".design"

_SAFE_NAME = re.compile(r"^[A-Za-z0-9_.-]+$")


def _sanitize(name: str, label: str) -> str:
    if not _SAFE_NAME.match(name):
        raise ValueError(f"Invalid {label}: {name!r} — must match {_SAFE_NAME.pattern}")
    return name


def _load_scenario(scenario_path: str) -> ScenarioDef:
    path = Path(scenario_path)
    if not path.exists():
        raise FileNotFoundError(f"Scenario file not found: {scenario_path}")
    data = yaml.safe_load(path.read_text())  # T-5.1-01
    return ScenarioDef.model_validate(data)


def _load_mocks(scenario_path: str, explicit_mocks: str | None) -> dict:
    if explicit_mocks:
        mocks_path = Path(explicit_mocks)
    else:
        p = Path(scenario_path)
        stem = p.name.replace(".scenario.yaml", "")
        mocks_path = p.parent / f"{stem}.mocks.yaml"
    if not mocks_path.exists():
        return {}
    return yaml.safe_load(mocks_path.read_text()) or {}  # T-5.1-01


def _write_trace(domain: str, scenario_path: str, steps: list) -> str:
    stem = Path(scenario_path).name.replace(".scenario.yaml", "")
    stem = re.sub(r"[^A-Za-z0-9_.-]", "_", stem)
    timestamp = datetime.now().strftime("%Y%m%d_%H%M%S_%f")
    traces_dir = _model_root() / "traces"
    traces_dir.mkdir(parents=True, exist_ok=True)
    out_path = traces_dir / f"{domain}_{stem}_{timestamp}.json"
    records = [_step_to_dict(s) for s in steps]
    out_path.write_text(json.dumps(records, indent=2, default=str))
    return str(out_path)


def _step_to_dict(step) -> dict:
    if is_dataclass(step):
        return {"type": step.__class__.__name__, **asdict(step)}
    if isinstance(step, dict):
        return dict(step)
    return {"type": type(step).__name__, "repr": repr(step)}


def _collect_final_states(ctx: SimulationContext) -> dict:
    out: dict[str, dict] = {}
    for class_name in ctx.registry.all_class_names():
        out[class_name] = {}
        for inst in ctx.registry.all_of(class_name):
            id_str = ",".join(f"{k}={v}" for k, v in sorted(inst["__instance_key__"]))
            state = ctx.registry.current_state(class_name, inst["__instance_key__"])
            out[class_name][id_str] = state
    return out


def _cleanup_generated_modules() -> None:
    for name in list(sys.modules):
        if name.startswith("mdf_generated_"):
            del sys.modules[name]


def simulate_domain(domain: str, scenario: str, mocks: str | None = None) -> dict:
    """MCP tool: load <domain>.mdfbundle and run the scenario against it."""
    errors: list[dict] = []
    trace_file = ""
    steps: list = []
    final_states: dict = {}
    tmpdir: Path | None = None

    try:
        domain_safe = _sanitize(domain, "domain")
        bundle_path = _model_root() / "bundles" / f"{domain_safe}.mdfbundle"
        if not bundle_path.exists():
            raise FileNotFoundError(f"Bundle not found: {bundle_path}")

        scenario_def = _load_scenario(scenario)
        bridge_mocks = _load_mocks(scenario, mocks)
        manifest, tmpdir = load_bundle(bundle_path)

        preflight_issues = check_multiplicity(scenario_def, manifest)
        if preflight_issues:
            for issue in preflight_issues:
                errors.append({"type": "PreflightError", "message": f"{issue.location}: {issue.message}"})
            trace_file = _write_trace(domain_safe, scenario, [])
            return {"total_steps": 0, "final_instance_states": {}, "errors": errors, "trace_file": trace_file}

        ctx = SimulationContext(manifest, bridge_mocks=bridge_mocks)
        for step in run_scenario(ctx, scenario_def, manifest):
            steps.append(step)
        final_states = _collect_final_states(ctx)

    except (BundleVersionError, BundleCorruptError) as e:
        errors.append({"type": type(e).__name__, "message": str(e)})
    except FileNotFoundError as e:
        errors.append({"type": "FileNotFoundError", "message": str(e)})
    except Exception as e:
        errors.append({"type": type(e).__name__, "message": str(e)})
    finally:
        _cleanup_generated_modules()
        if tmpdir is not None:
            shutil.rmtree(tmpdir, ignore_errors=True)

    if not trace_file:
        trace_file = _write_trace(domain if _SAFE_NAME.match(domain or "") else "unknown", scenario, steps)

    return {
        "total_steps": len(steps),
        "final_instance_states": final_states,
        "errors": errors,
        "trace_file": trace_file,
    }


def simulate_class(class_name: str, scenario: str, mocks: str | None = None) -> dict:
    """MCP tool: isolated single-class simulation. Locates the bundle that defines class_name."""
    errors: list[dict] = []
    try:
        class_safe = _sanitize(class_name, "class_name")
        bundle_path = _find_bundle_for_class(class_safe)
        if bundle_path is None:
            return {
                "total_steps": 0,
                "final_instance_states": {},
                "errors": [{"type": "ClassNotFound", "message": f"No bundle contains class {class_name}"}],
                "trace_file": "",
            }
        # Delegate to simulate_domain with the containing domain
        domain = bundle_path.stem  # <domain>.mdfbundle
        return simulate_domain(domain, scenario, mocks=mocks)
    except Exception as e:
        errors.append({"type": type(e).__name__, "message": str(e)})
        return {"total_steps": 0, "final_instance_states": {}, "errors": errors, "trace_file": ""}


def _find_bundle_for_class(class_name: str) -> Path | None:
    """Scan .design/bundles/ for a bundle whose generated/ dir contains <class_name>.py."""
    import zipfile
    bundles_dir = _model_root() / "bundles"
    if not bundles_dir.exists():
        return None
    for bundle in sorted(bundles_dir.glob("*.mdfbundle")):
        try:
            with zipfile.ZipFile(bundle) as zf:
                if f"generated/{class_name}.py" in zf.namelist():
                    return bundle
        except zipfile.BadZipFile:
            continue
    return None
```
