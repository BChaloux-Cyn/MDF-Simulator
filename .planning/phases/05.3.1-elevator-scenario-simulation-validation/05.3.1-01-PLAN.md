**Fix F: Update codegen transition table rendering for list format**

In `_render_transition_table`, update to handle list-valued entries. Each `(state, event)` key maps to a list of entries. For rendering, produce a list literal per key:

```python
def _render_transition_table(
    transition_table: dict,
    action_fn_map: dict[tuple[str, str], str],
    guard_fn_map: dict[tuple[str, str], str],
) -> str:
    lines = ["TRANSITION_TABLE: dict = {"]
    for key in sorted(transition_table.keys(), key=lambda k: (k[0], k[1])):
        state, event = key
        entries = transition_table[key]
        entry_strs = []
        for idx, entry in enumerate(entries):
            next_state = entry["next_state"]
            next_state_repr = repr(next_state)
            action_fn_name = action_fn_map.get(key)
            guard_fn_name = guard_fn_map.get(key)
            # For guard siblings: use index-qualified key for second+ entries
            if idx > 0:
                guard_fn_name = guard_fn_map.get((state, event, idx))
            action_ref = action_fn_name if action_fn_name else "None"
            guard_ref = guard_fn_name if guard_fn_name else "None"
            entry_strs.append(
                f'{{"next_state": {next_state_repr}, "action_fn": {action_ref}, "guard_fn": {guard_ref}}}'
            )
        if len(entry_strs) == 1:
            lines.append(f'    ({state!r}, {event!r}): [{entry_strs[0]}],')
        else:
            lines.append(f'    ({state!r}, {event!r}): [')
            for es in entry_strs:
                lines.append(f'        {es},')
            lines.append('    ],')
    lines.append("}")
    return "\n".join(lines)
```

**Fix G: Update scheduler dispatch for list-valued transition entries (engine/scheduler.py)**

In `_process_event`, replace the single-entry lookup (around line 389):

FROM:
```python
entry = ttable.get((curr_state, event.event_type))
if entry is None:
    yield ErrorMicroStep(error_kind="cant_happen", ...)
    return
next_state = entry.get("next_state")
action_fn = entry.get("action_fn")
guard_fn = entry.get("guard_fn")
```

TO:
```python
entries = ttable.get((curr_state, event.event_type))
if entries is None:
    yield ErrorMicroStep(error_kind="cant_happen", ...)
    return

# Evaluate guard chain: pick first entry whose guard passes (or unguarded fallback)
entry = None
for candidate in entries:
    guard_fn = candidate.get("guard_fn")
    if guard_fn is None:
        # Unguarded entry — use as default if no guarded entry matches
        if entry is None:
            entry = candidate
        continue
    try:
        instance = self._registry.lookup(concrete_class, concrete_id)
        result = bool(guard_fn(instance, dict(event.args)))
    except Exception as exc:
        yield ErrorMicroStep(
            error_kind="guard_error",
            message=str(exc),
            context={"class": concrete_class, "event_type": event.event_type},
        )
        return
    yield GuardEvaluated(
        expression=getattr(guard_fn, "__name__", "<guard>"),
        result=result,
        variable_values={},
    )
    if result:
        entry = candidate
        break

if entry is None:
    # All guards failed and no unguarded fallback
    return

next_state = entry.get("next_state")
action_fn = entry.get("action_fn")
```

Remove the standalone guard evaluation block that was previously below the entry lookup (since guard evaluation is now integrated into the entry selection loop above).

Also update `_process_creation` if it reads from transition_table: ensure it handles list format (e.g., `entries = ttable.get((None, "__creation__"))` becomes `entries = ttable.get(...)` then `entry = entries[0] if entries else None`).

Also handle event-ignored detection: check if ALL entries in the list are event-ignored (all have `next_state=None`, `action_fn=None`, `guard_fn=None`).
